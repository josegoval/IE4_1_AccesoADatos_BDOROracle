	-- ALUMNO --

CREATE OR REPLACE FUNCTION DELETE_ALUMNO (
	DNI_IN IN VARCHAR2
	) RETURN NUMBER
IS
BEGIN
	DELETE FROM TALUMNOS A WHERE A.DNI = DNI_IN;
	RETURN 1;
EXCEPTION
	WHEN OTHERS THEN
		RETURN 0;
END;


CREATE OR REPLACE FUNCTION DELETE_ALUMNO_ASIGNATURA (
	DNI_ALUMNO_IN IN VARCHAR2, 
	COD_ASIG_IN IN NUMBER
	)
	RETURN NUMBER
IS
BEGIN
	DELETE FROM TABLE(SELECT A.TNOTAS FROM TALUMNOS A WHERE A.DNI = DNI_ALUMNO_IN) N
		WHERE N.ASIGNATURA.COD_ASIG = COD_ASIG_IN;
	RETURN 1;
EXCEPTION
	WHEN OTHERS THEN
		RETURN 0;		
END;
	
	-- ASIGNATURA --

CREATE OR REPLACE FUNCTION DELETE_ASIGNATURA (
	COD_ASIG_IN IN NUMBER
	) RETURN NUMBER
IS
BEGIN
	DELETE FROM TASIGNATURAS A WHERE A.COD_ASIG = COD_ASIG_IN;
	RETURN 1;
EXCEPTION
	WHEN OTHERS THEN
		RETURN 0;	
END;


	-- CURSO --

	ID_CURSO_IN NUMBER

END;
